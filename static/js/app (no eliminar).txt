"use strict";

const App = {

    tasks: [],

    currentView: "dashboard",

    init() {

        this.cache();

        this.bindEvents();

        this.showDate();

        this.changeView("dashboard");

        this.renderTable();

    },

    cache() {

        this.views = document.querySelectorAll(".view");
        this.menuButtons = document.querySelectorAll(".menu-item");

        this.modal = document.getElementById("taskModal");

        this.form = document.getElementById("taskForm");

        this.table = document.querySelector("#taskTable tbody");

        this.search = document.getElementById("txtSearch");

        this.btnNew = document.getElementById("btnNewTask");

        this.btnCancel = document.getElementById("btnCancel");

    },

    bindEvents() {

        this.menuButtons.forEach(btn => {

            btn.addEventListener("click", () => {
                this.changeView(btn.dataset.view);
            });

        });

        this.btnNew.addEventListener("click", () => this.openModal());

        this.btnCancel.addEventListener("click", () => this.closeModal());

        this.form.addEventListener("submit", (e) => {
            e.preventDefault();
            this.saveTask();
        });

        this.search.addEventListener("input", () => {
            this.renderTable(this.search.value);
        });

    },

    changeView(view) {

        this.currentView = view;

        this.views.forEach(v => v.classList.remove("active"));

        document.getElementById(view).classList.add("active");

        this.menuButtons.forEach(b => b.classList.remove("active"));

        document.querySelector(`[data-view="${view}"]`).classList.add("active");

    },

    showDate() {

        const d = new Date();

        document.getElementById("currentDate").textContent =
            d.toLocaleDateString("es-ES", {
                weekday: "long",
                day: "numeric",
                month: "long",
                year: "numeric"
            });

    },

    openModal() {

        this.form.reset();

        this.modal.classList.remove("hidden");

    },

    closeModal() {

        this.modal.classList.add("hidden");

    },

    saveTask() {

        const task = {

            id: Date.now(),

            titulo: document.getElementById("titulo").value,

            descripcion: document.getElementById("descripcion").value,

            responsable: document.getElementById("responsable").value,

            prioridad: document.getElementById("prioridad").value,

            fecha: document.getElementById("fechaPrevista").value,

            estado: "Pendiente",

            avance: 0

        };

        this.tasks.push(task);

        this.renderTable();

        this.updateDashboard();

        this.closeModal();

    },

    renderTable(filter = "") {

        this.table.innerHTML = "";

        let data = this.tasks;

        if (filter.trim() !== "") {

            const text = filter.toLowerCase();

            data = data.filter(t =>
                t.titulo.toLowerCase().includes(text) ||
                t.responsable.toLowerCase().includes(text)
            );

        }

        data.forEach(task => {

            const tr = document.createElement("tr");

            tr.innerHTML = `
                <td>${task.id}</td>
                <td>${task.titulo}</td>
                <td>${task.responsable}</td>
                <td>${task.prioridad}</td>
                <td>${task.estado}</td>
                <td>-</td>
                <td>${task.fecha}</td>
                <td>${task.avance}%</td>
                <td>
                    <button onclick="alert('Edición en desarrollo')">✏️</button>
                </td>
            `;

            this.table.appendChild(tr);

        });

    },

    updateDashboard() {

        document.getElementById("totalTasks").textContent = this.tasks.length;

        document.getElementById("pendingTasks").textContent =
            this.tasks.filter(t => t.estado === "Pendiente").length;

        document.getElementById("runningTasks").textContent =
            this.tasks.filter(t => t.estado === "En curso").length;

        document.getElementById("finishedTasks").textContent =
            this.tasks.filter(t => t.estado === "Finalizada").length;

    }

};

window.addEventListener("DOMContentLoaded", () => App.init());

Object.assign(App, {

    editingId: null,

    editTask(id) {

        const task = this.tasks.find(t => t.id === id);

        if (!task) return;

        this.editingId = id;

        document.getElementById("titulo").value = task.titulo;
        document.getElementById("descripcion").value = task.descripcion;
        document.getElementById("responsable").value = task.responsable;
        document.getElementById("prioridad").value = task.prioridad;
        document.getElementById("fechaPrevista").value = task.fecha;

        this.openModal();

    },

    deleteTask(id) {

        if (!confirm("¿Eliminar esta tarea?"))
            return;

        this.tasks = this.tasks.filter(t => t.id !== id);

        this.renderTable();
        this.updateDashboard();

    },

    saveTask() {

        const nueva = {
            id: this.editingId ?? Date.now(),
              titulo: document.getElementById("titulo").value.trim(),
              descripcion: document.getElementById("descripcion").value.trim(),
              responsable: document.getElementById("responsable").value.trim(),
              prioridad: document.getElementById("prioridad").value,
              fecha: document.getElementById("fechaPrevista").value,
              estado: "Pendiente",
              avance: 0
        };

        if (nueva.titulo === "") {
            alert("Debe indicar un título.");
            return;
        }

        if (this.editingId !== null) {

            const pos = this.tasks.findIndex(t => t.id === this.editingId);

            if (pos >= 0)
                this.tasks[pos] = nueva;

            this.editingId = null;

        } else {

            this.tasks.push(nueva);

        }

        this.renderTable();
        this.updateDashboard();
        this.closeModal();

    },

    renderTable(filter = "") {

        this.table.innerHTML = "";

        let lista = this.tasks;

        if (filter.trim() !== "") {

            filter = filter.toLowerCase();

            lista = lista.filter(t =>
            t.titulo.toLowerCase().includes(filter) ||
            t.descripcion.toLowerCase().includes(filter) ||
            t.responsable.toLowerCase().includes(filter)
            );

        }

        lista.forEach(task => {

            const tr = document.createElement("tr");

            tr.innerHTML = `
            <td>${task.id}</td>
            <td>${task.titulo}</td>
            <td>${task.responsable}</td>
            <td>${task.prioridad}</td>
            <td>${task.estado}</td>
            <td>-</td>
            <td>${task.fecha}</td>
            <td>${task.avance}%</td>
            <td>
            <button class="btnEdit">✏️</button>
            <button class="btnDelete">🗑️</button>
            </td>
            `;

            tr.querySelector(".btnEdit")
            .addEventListener("click", () => this.editTask(task.id));

            tr.querySelector(".btnDelete")
            .addEventListener("click", () => this.deleteTask(task.id));

            this.table.appendChild(tr);

        });

    }

});

Object.assign(App, {

    storageKey: "todo_tasks_v01",

    loadTasks() {

        try {

            const saved = localStorage.getItem(this.storageKey);

            this.tasks = saved ? JSON.parse(saved) : [];

        } catch (error) {

            console.error("No se pudieron cargar las tareas:", error);

            this.tasks = [];

        }

    },

    saveTasks() {

        try {

            localStorage.setItem(
                this.storageKey,
                JSON.stringify(this.tasks)
            );

        } catch (error) {

            console.error("No se pudieron guardar las tareas:", error);

        }

    },

    ensureTaskControls() {

        const section = document.getElementById("tasks");

        if (!section || document.getElementById("taskFilters"))
            return;

        const controls = document.createElement("div");

        controls.id = "taskFilters";

        controls.innerHTML = `
        <select id="filterStatus">
        <option value="">Todos los estados</option>
        <option value="Pendiente">Pendiente</option>
        <option value="En curso">En curso</option>
        <option value="Bloqueada">Bloqueada</option>
        <option value="Finalizada">Finalizada</option>
        <option value="Cancelada">Cancelada</option>
        </select>

        <select id="filterPriority">
        <option value="">Todas las prioridades</option>
        <option value="Baja">Baja</option>
        <option value="Media">Media</option>
        <option value="Alta">Alta</option>
        <option value="Crítica">Crítica</option>
        </select>

        <select id="sortTasks">
        <option value="id-desc">Más recientes</option>
        <option value="id-asc">Más antiguas</option>
        <option value="title-asc">Título A-Z</option>
        <option value="date-asc">Fecha prevista</option>
        <option value="progress-desc">Mayor avance</option>
        </select>
        `;

        section.insertBefore(
            controls,
            document.getElementById("taskTable")
        );

        document.getElementById("filterStatus")
        .addEventListener("change", () => this.renderTable());

        document.getElementById("filterPriority")
        .addEventListener("change", () => this.renderTable());

        document.getElementById("sortTasks")
        .addEventListener("change", () => this.renderTable());

    },

    getFilteredTasks(filter = "") {

        let list = [...this.tasks];

        const text = filter.trim().toLowerCase();

        if (text !== "") {

            list = list.filter(task =>
            task.titulo.toLowerCase().includes(text) ||
            task.descripcion.toLowerCase().includes(text) ||
            task.responsable.toLowerCase().includes(text)
            );

        }

        const statusFilter =
        document.getElementById("filterStatus")?.value || "";

        const priorityFilter =
        document.getElementById("filterPriority")?.value || "";

        if (statusFilter) {

            list = list.filter(
                task => task.estado === statusFilter
            );

        }

        if (priorityFilter) {

            list = list.filter(
                task => task.prioridad === priorityFilter
            );

        }

        const sortMode =
        document.getElementById("sortTasks")?.value || "id-desc";

        list.sort((a, b) => {

            switch (sortMode) {

                case "id-asc":
                    return a.id - b.id;

                case "title-asc":
                    return a.titulo.localeCompare(
                        b.titulo,
                        "es",
                        { sensitivity: "base" }
                    );

                case "date-asc":
                    return (a.fecha || "9999-12-31")
                    .localeCompare(b.fecha || "9999-12-31");

                case "progress-desc":
                    return Number(b.avance) - Number(a.avance);

                case "id-desc":
                default:
                    return b.id - a.id;

            }

        });

        return list;

    },

    changeTaskStatus(id, newStatus) {

        const task = this.tasks.find(item => item.id === id);

        if (!task)
            return;

        task.estado = newStatus;

        if (newStatus === "Finalizada") {

            task.avance = 100;

        } else if (newStatus === "Pendiente" && task.avance === 100) {

            task.avance = 0;

        }

        this.saveTasks();
        this.renderTable();
        this.updateDashboard();

    },

    changeTaskProgress(id, progress) {

        const task = this.tasks.find(item => item.id === id);

        if (!task)
            return;

        const value = Math.max(
            0,
            Math.min(100, Number(progress) || 0)
        );

        task.avance = value;

        if (value === 100) {

            task.estado = "Finalizada";

        } else if (
            value > 0 &&
            task.estado === "Pendiente"
        ) {

            task.estado = "En curso";

        }

        this.saveTasks();
        this.renderTable();
        this.updateDashboard();

    },

    renderTable(filter = "") {

        this.table.innerHTML = "";

        const list = this.getFilteredTasks(
            filter || this.search.value
        );

        list.forEach(task => {

            const tr = document.createElement("tr");

            tr.innerHTML = `
            <td>${task.id}</td>
            <td>${task.titulo}</td>
            <td>${task.responsable || "-"}</td>
            <td>${task.prioridad}</td>
            <td>
            <select class="statusSelect">
            <option ${task.estado === "Pendiente" ? "selected" : ""}>Pendiente</option>
            <option ${task.estado === "En curso" ? "selected" : ""}>En curso</option>
            <option ${task.estado === "Bloqueada" ? "selected" : ""}>Bloqueada</option>
            <option ${task.estado === "Finalizada" ? "selected" : ""}>Finalizada</option>
            <option ${task.estado === "Cancelada" ? "selected" : ""}>Cancelada</option>
            </select>
            </td>
            <td>-</td>
            <td>${task.fecha || "-"}</td>
            <td>
            <div class="progressWrapper">
            <progress
            value="${task.avance}"
            max="100">
            </progress>

            <input
            class="progressInput"
            type="number"
            min="0"
            max="100"
            value="${task.avance}"
            >
            </div>
            </td>
            <td>
            <button class="btnEdit" title="Editar">✏️</button>
            <button class="btnDelete" title="Eliminar">🗑️</button>
            </td>
            `;

            tr.querySelector(".statusSelect")
            .addEventListener("change", event => {
                this.changeTaskStatus(
                    task.id,
                    event.target.value
                );
            });

            tr.querySelector(".progressInput")
            .addEventListener("change", event => {
                this.changeTaskProgress(
                    task.id,
                    event.target.value
                );
            });

            tr.querySelector(".btnEdit")
            .addEventListener("click", () => {
                this.editTask(task.id);
            });

            tr.querySelector(".btnDelete")
            .addEventListener("click", () => {
                this.deleteTask(task.id);
            });

            this.table.appendChild(tr);

        });

    },

    saveTask() {

        const task = {
            id: this.editingId ?? Date.now(),
              titulo: document.getElementById("titulo").value.trim(),
              descripcion: document.getElementById("descripcion").value.trim(),
              responsable: document.getElementById("responsable").value.trim(),
              prioridad: document.getElementById("prioridad").value,
              fecha: document.getElementById("fechaPrevista").value,
              estado: "Pendiente",
              avance: 0
        };

        if (task.titulo === "") {

            alert("Debe indicar un título.");

            return;

        }

        if (this.editingId !== null) {

            const position = this.tasks.findIndex(
                item => item.id === this.editingId
            );

            if (position >= 0) {

                task.estado = this.tasks[position].estado;
                task.avance = this.tasks[position].avance;

                this.tasks[position] = task;

            }

            this.editingId = null;

        } else {

            this.tasks.push(task);

        }

        this.saveTasks();
        this.renderTable();
        this.updateDashboard();
        this.closeModal();

    },

    deleteTask(id) {

        if (!confirm("¿Eliminar esta tarea?"))
            return;

        this.tasks = this.tasks.filter(
            task => task.id !== id
        );

        this.saveTasks();
        this.renderTable();
        this.updateDashboard();

    }

});

Object.assign(App, {

    calendarDate: new Date(),

              ensureCalendarLayout() {

                  const container = document.getElementById("calendarContainer");

                  if (!container || document.getElementById("calendarGrid"))
                      return;

                  container.innerHTML = `
                  <div class="calendarHeader">
                  <button id="calendarPrev" type="button">◀</button>

                  <h3 id="calendarTitle"></h3>

                  <button id="calendarToday" type="button">Hoy</button>

                  <button id="calendarNext" type="button">▶</button>
                  </div>

                  <div class="calendarWeekdays">
                  <span>Lunes</span>
                  <span>Martes</span>
                  <span>Miércoles</span>
                  <span>Jueves</span>
                  <span>Viernes</span>
                  <span>Sábado</span>
                  <span>Domingo</span>
                  </div>

                  <div id="calendarGrid" class="calendarGrid"></div>

                  <div id="calendarDayPanel" class="calendarDayPanel hidden">
                  <div class="calendarDayPanelHeader">
                  <h3 id="calendarSelectedDate"></h3>
                  <button id="calendarClosePanel" type="button">✕</button>
                  </div>

                  <div id="calendarSelectedTasks"></div>

                  <button id="calendarCreateTask" type="button">
                  Nueva tarea para este día
                  </button>
                  </div>
                  `;

                  document.getElementById("calendarPrev")
                  .addEventListener("click", () => {

                      this.calendarDate.setMonth(
                          this.calendarDate.getMonth() - 1
                      );

                      this.renderCalendar();

                  });

                  document.getElementById("calendarNext")
                  .addEventListener("click", () => {

                      this.calendarDate.setMonth(
                          this.calendarDate.getMonth() + 1
                      );

                      this.renderCalendar();

                  });

                  document.getElementById("calendarToday")
                  .addEventListener("click", () => {

                      this.calendarDate = new Date();

                      this.renderCalendar();

                  });

                  document.getElementById("calendarClosePanel")
                  .addEventListener("click", () => {

                      document.getElementById("calendarDayPanel")
                      .classList.add("hidden");

                  });

              },

              renderCalendar() {

                  this.ensureCalendarLayout();

                  const grid = document.getElementById("calendarGrid");
                  const title = document.getElementById("calendarTitle");

                  if (!grid || !title)
                      return;

                  grid.innerHTML = "";

                  const year = this.calendarDate.getFullYear();
                  const month = this.calendarDate.getMonth();

                  title.textContent = new Intl.DateTimeFormat(
                      "es-ES",
                      {
                          month: "long",
                          year: "numeric"
                      }
                  ).format(new Date(year, month, 1));

                  const firstDay = new Date(year, month, 1);

                  let startOffset = firstDay.getDay() - 1;

                  if (startOffset < 0)
                      startOffset = 6;

                  const daysInMonth =
                  new Date(year, month + 1, 0).getDate();

                  const previousMonthDays =
                  new Date(year, month, 0).getDate();

                  const totalCells = 42;

                  for (let index = 0; index < totalCells; index++) {

                      let dayNumber;
                      let cellDate;
                      let outsideMonth = false;

                      if (index < startOffset) {

                          dayNumber =
                          previousMonthDays - startOffset + index + 1;

                          cellDate =
                          new Date(year, month - 1, dayNumber);

                          outsideMonth = true;

                      } else if (
                          index >= startOffset + daysInMonth
                      ) {

                          dayNumber =
                          index - startOffset - daysInMonth + 1;

                          cellDate =
                          new Date(year, month + 1, dayNumber);

                          outsideMonth = true;

                      } else {

                          dayNumber = index - startOffset + 1;

                          cellDate =
                          new Date(year, month, dayNumber);

                      }

                      const isoDate = this.toISODate(cellDate);

                      const tasksForDay = this.tasks.filter(
                          task => task.fecha === isoDate
                      );

                      const cell = document.createElement("button");

                      cell.type = "button";
                      cell.className = "calendarCell";

                      if (outsideMonth)
                          cell.classList.add("outsideMonth");

                      if (isoDate === this.toISODate(new Date()))
                          cell.classList.add("today");

                      cell.innerHTML = `
                      <span class="calendarDayNumber">
                      ${dayNumber}
                      </span>

                      <div class="calendarTaskList">
                      ${tasksForDay.slice(0, 3).map(task => `
                          <span
                          class="calendarTask calendarStatus-${this.normaliseCssValue(task.estado)}"
                          title="${this.escapeHtml(task.titulo)}"
                          >
                          ${this.escapeHtml(task.titulo)}
                          </span>
                          `).join("")}

                          ${
                              tasksForDay.length > 3
                              ? `<small>+${tasksForDay.length - 3} más</small>`
                              : ""
                          }
                          </div>
                          `;

                          cell.addEventListener("click", () => {

                              this.showCalendarDay(isoDate);

                          });

                          grid.appendChild(cell);

                  }

              },

              showCalendarDay(isoDate) {

                  const panel =
                  document.getElementById("calendarDayPanel");

                  const title =
                  document.getElementById("calendarSelectedDate");

                  const list =
                  document.getElementById("calendarSelectedTasks");

                  const createButton =
                  document.getElementById("calendarCreateTask");

                  if (!panel || !title || !list || !createButton)
                      return;

                  const selectedDate =
                  new Date(`${isoDate}T12:00:00`);

                  title.textContent = new Intl.DateTimeFormat(
                      "es-ES",
                      {
                          weekday: "long",
                          day: "numeric",
                          month: "long",
                          year: "numeric"
                      }
                  ).format(selectedDate);

                  const tasks = this.tasks.filter(
                      task => task.fecha === isoDate
                  );

                  if (tasks.length === 0) {

                      list.innerHTML = `
                      <p class="calendarEmpty">
                      No hay tareas previstas para este día.
                      </p>
                      `;

                  } else {

                      list.innerHTML = tasks.map(task => `
                      <article class="calendarDayTask">
                      <div>
                      <strong>${this.escapeHtml(task.titulo)}</strong>
                      <small>
                      ${this.escapeHtml(task.responsable || "Sin responsable")}
                      </small>
                      </div>

                      <div>
                      <span>${this.escapeHtml(task.estado)}</span>
                      <button
                      type="button"
                      data-edit-calendar-task="${task.id}"
                      >
                      Editar
                      </button>
                      </div>
                      </article>
                      `).join("");

                      list.querySelectorAll(
                          "[data-edit-calendar-task]"
                      ).forEach(button => {

                          button.addEventListener("click", () => {

                              this.editTask(
                                  Number(button.dataset.editCalendarTask)
                              );

                          });

                      });

                  }

                  createButton.onclick = () => {

                      this.openModal();

                      document.getElementById("fechaPrevista").value =
                      isoDate;

                  };

                  panel.classList.remove("hidden");

              },

              toISODate(date) {

                  const year = date.getFullYear();

                  const month = String(
                      date.getMonth() + 1
                  ).padStart(2, "0");

                  const day = String(
                      date.getDate()
                  ).padStart(2, "0");

                  return `${year}-${month}-${day}`;

              },

              normaliseCssValue(value) {

                  return String(value || "")
                  .normalize("NFD")
                  .replace(/[\u0300-\u036f]/g, "")
                  .toLowerCase()
                  .replace(/\s+/g, "-")
                  .replace(/[^a-z0-9-]/g, "");

              },

              escapeHtml(value) {

                  return String(value ?? "")
                  .replaceAll("&", "&amp;")
                  .replaceAll("<", "&lt;")
                  .replaceAll(">", "&gt;")
                  .replaceAll('"', "&quot;")
                  .replaceAll("'", "&#039;");

              }

});

Object.assign(App, {

    calculateStatistics() {

        const stats = {
            total: this.tasks.length,
            pending: 0,
            running: 0,
            blocked: 0,
            finished: 0,
            cancelled: 0,
            progress: 0,
            overdue: 0
        };

        const today = this.toISODate(new Date());

        this.tasks.forEach(task => {

            switch (task.estado) {
                case "Pendiente": stats.pending++; break;
                case "En curso": stats.running++; break;
                case "Bloqueada": stats.blocked++; break;
                case "Finalizada": stats.finished++; break;
                case "Cancelada": stats.cancelled++; break;
            }

            stats.progress += Number(task.avance || 0);

            if (
                task.fecha &&
                task.estado !== "Finalizada" &&
                task.fecha < today
            ) {
                stats.overdue++;
            }

        });

        stats.averageProgress =
        stats.total === 0
        ? 0
        : Math.round(stats.progress / stats.total);

        return stats;

    },

    renderDashboard() {

        const s = this.calculateStatistics();

Object.assign(App, {

    ensureExtendedFields() {

        this.tasks.forEach(task => {
            task.tags ??= [];
            task.comments ??= [];
            task.favorite ??= false;
            task.attachments ??= [];
            task.reminder ??= "";
        });

    },

    toggleFavorite(id) {

        const task = this.tasks.find(t => t.id === id);
        if (!task) return;

        task.favorite = !task.favorite;

        this.saveTasks();
        this.renderTable();
    },

    addTag(id, tag) {

        const task = this.tasks.find(t => t.id === id);
        if (!task) return;

        tag = String(tag).trim();

        if (!tag)
            return;

        task.tags ??= [];

        if (!task.tags.includes(tag))
            task.tags.push(tag);

        this.saveTasks();
    },

    addComment(id, text) {

        const task = this.tasks.find(t => t.id === id);
        if (!task) return;

        task.comments ??= [];

        task.comments.push({
            date: new Date().toISOString(),
                           text: text
        });

        this.saveTasks();
    },

    setReminder(id, date) {

        const task = this.tasks.find(t => t.id === id);
        if (!task) return;

        task.reminder = date;

        this.saveTasks();
    }

});

Object.assign(App, {

    selectedTasks: new Set(),

              toggleSelection(id) {

                  if (this.selectedTasks.has(id))
                      this.selectedTasks.delete(id);
                  else
                      this.selectedTasks.add(id);

              },

              clearSelection() {
                  this.selectedTasks.clear();
              },

              exportCSV() {

                  const headers = [
                      "ID","Título","Responsable","Prioridad",
                      "Estado","Fecha","Avance"
                  ];

                  const rows = this.tasks.map(t => [
                      t.id,
                      `"${(t.titulo||"").replace(/"/g,'""')}"`,
                                              `"${(t.responsable||"").replace(/"/g,'""')}"`,
                                              t.prioridad,
                                              t.estado,
                                              t.fecha,
                                              t.avance
                  ].join(","));

                  const csv = [headers.join(","), ...rows].join("\n");

                  const blob = new Blob([csv], {type:"text/csv;charset=utf-8"});
                  const url = URL.createObjectURL(blob);

                  const a = document.createElement("a");
                  a.href = url;
                  a.download = "tareas.csv";
                  a.click();

                  URL.revokeObjectURL(url);

              },

              bulkChangeStatus(status){

                  this.tasks.forEach(task=>{
                      if(this.selectedTasks.has(task.id))
                          task.estado=status;
                  });

                  this.saveTasks();
                  this.renderTable();
                  this.updateDashboard();

              },

              bulkDelete(){

                  if(this.selectedTasks.size===0)
                      return;

                  if(!confirm("¿Eliminar las tareas seleccionadas?"))
                      return;

                  this.tasks=this.tasks.filter(
                      t=>!this.selectedTasks.has(t.id)
                  );

                  this.clearSelection();
                  this.saveTasks();
                  this.renderTable();
                  this.updateDashboard();

              }

});

    apiBase:"/api",

    async api(url,options={}){

        const cfg={
            headers:{
                "Content-Type":"application/json"
            },
            ...options
        };

        const response=await fetch(
            this.apiBase+url,
            cfg
        );

        if(!response.ok){
            throw new Error(
                "Error HTTP "+response.status
            );
        }

        const type=response.headers.get("content-type")||"";

        if(type.includes("application/json"))
            return await response.json();

        return await response.text();

    },

    },

            );

        }

    },

Object.assign(App, {

    async refreshFromServer() {

        try {

            const data = await this.api("/tasks");

            if (!Array.isArray(data))
                return;

            this.tasks = data;

            this.ensureExtendedFields?.();

            this.renderTable();
            this.updateDashboard();

            if (this.currentView === "calendar") {
                this.renderCalendar?.();
            }

        } catch (e) {

            console.error(e);
            this.showNotification?.(
                "No se pudieron cargar las tareas.",
                "error"
            );

        }

    },

    async createTaskServer(task) {

        const created = await this.api("/tasks", {
            method: "POST",
            body: JSON.stringify(task)
        });

        await this.refreshFromServer();

        return created;

    },

    async updateTaskServer(id, task, partial = true) {

        const method = partial ? "PATCH" : "PUT";

        const updated = await this.api("/tasks/" + id, {
            method,
            body: JSON.stringify(task)
        });

        await this.refreshFromServer();

        return updated;

    },

    async deleteTaskServer(id) {

        await this.api("/tasks/" + id, {
            method: "DELETE"
        });

        await this.refreshFromServer();

    }

});

const _saveTask = App.saveTask.bind(App);

App.saveTask = async function () {

    const editing = this.editingId;

    _saveTask();

    const last = this.tasks[this.tasks.length - 1];

    try {

        if (editing !== null) {

            await this.updateTaskServer(
                editing,
                last,
                true
            );

        } else {

            await this.createTaskServer(last);

        }

        this.showNotification?.(
            "Cambios guardados.",
            "success"
        );

    } catch (e) {

        console.error(e);

        this.showNotification?.(
            "Error al guardar.",
            "error"
        );

    }

};

        this.showNotification?.(

            "success"
        );

    }catch(e){

        console.error(e);

        this.showNotification?.(
            "No se pudo eliminar.",

        );

    }

};
